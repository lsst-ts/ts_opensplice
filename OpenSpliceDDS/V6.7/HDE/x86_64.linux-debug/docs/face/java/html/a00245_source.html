163  public void receiveMessage(long timeout, LongHolder transaction_id,
164  Holder<TYPE> message, int message_size,
165  RETURN_CODE_TYPEHolder return_code) {
166  try {
167  if (timeout == FACE.INF_TIME_VALUE.value) {
168  timeout = Long.MAX_VALUE;
169  }
170  this.waitset.waitForConditions(timeout, TimeUnit.NANOSECONDS);
171  Iterator<TYPE> samples = this.dataReader.take(this.selector);
172  Sample<TYPE> sample = samples.next();
173  this.setLastMessageValidity(VALIDITY_TYPE.VALID);
174 
175  if (sample != null) {
176  transaction_id.value = this.transactionid.incrementAndGet();
177  message.value = sample.getData();
178  return_code.value = RETURN_CODE_TYPE.NO_ERROR;
179  } else {
180  Logger.getInstance()
181  .log("receiveMessage took no data even though waitset did trigger.",
182  Level.SEVERE);
183  return_code.value = RETURN_CODE_TYPE.INVALID_CONFIG;
184  }
185 
186  } catch (TimeoutException e) {
187  Logger.getInstance().log("receiveMessage timed out.", Level.FINE);
188  return_code.value = RETURN_CODE_TYPE.TIMED_OUT;
189  this.setLastMessageValidity(VALIDITY_TYPE.VALID); /* Timeout is valid */
190  } catch (AlreadyClosedException e) {
191  this.setLastMessageValidity(VALIDITY_TYPE.INVALID);
192  } catch (Exception exc) {
193  Logger.getInstance().log(exc.getMessage(), Level.SEVERE);
