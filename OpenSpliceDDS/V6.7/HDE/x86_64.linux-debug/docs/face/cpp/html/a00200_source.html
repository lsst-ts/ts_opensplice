31 /*
32  * This class is mainly a map of <type name, typed create function>
33  * pairs. The 'type name' is the key, while the 'typed create function'
34  * can be used to create Connection<TYPE> objects.
35  *
36  * When the type is found in the map, then the 'second' part of
37  * the found pair is actually the templated create function
38  * "newConnection<TYPE>".
39  * Meaning that the it->second() will return a new instance of
40  * the Connection<TYPE> cast to AnyConnection.
41  */
42 class VORTEX_FACE_API ConnectionFactory

45  typedef std::map<std::string, AnyConnection*(*)()> CF_MAP_TYPE;

47  /*
48  * This function can be used to create a Connection<TYPE>:

51  * typedConnection->doTypedStuff();

84 /*
85  * This class will add a templated create function "newConnection<TYPE>"
86  * to the ConnectionFactory map with the type name as key.
87  *
88  * This class will be a static member variable of the specialized typed
89  * connections which means that typed creation function is registered
90  * automatically for the given typed connection.
91  *
92  * Example for Space::Type1:
93  *
94  * class SpaceType1ConnectionRegistration
95  * {
96  * static ConnectionFactoryTypeRegister< Connection<Space::Type1> > reg;
97  * };
98  * ConnectionFactoryTypeRegister< Connection<Space::Type1> > SpaceType1ConnectionRegistration::reg("Space::Type1");
99  *
100  * The initialization of SpaceType1ConnectionRegistration::reg; will result
101  * in a <type name, typed create function> pair in the ConnectionFactory map.
102  */
103 template<typename TYPE>
104 struct ConnectionFactoryTypeRegister : ConnectionFactory {
105  /* When this class is a static object, then this constructor is called at
106  * startup. This means that the related type is registered at startup as well */
107  ConnectionFactoryTypeRegister(std::string const& typeName) {
108  /* Add the <type name, typed create function> pair to the factory. */
109  getMap()->insert(std::pair<std::string, AnyConnection*(*)()>(typeName, &newConnection<TYPE>));
110  }
111 };
