42  private static void checkReturnCode(int retCode,
43  OsplServiceEnvironment environment, String message,
44  boolean withTimeOut) throws TimeOutExceptionImpl {
45  switch (retCode) {
46  case DDS.RETCODE_PRECONDITION_NOT_MET.value:
47  throw new PreconditionNotMetExceptionImpl(environment,
48  getErrorMessage(retCode, message));
49  case DDS.RETCODE_OUT_OF_RESOURCES.value:
50  throw new OutOfResourcesExceptionImpl(environment, getErrorMessage(
51  retCode, message));
52  case DDS.RETCODE_ALREADY_DELETED.value:
53  throw new AlreadyClosedExceptionImpl(environment, getErrorMessage(
54  retCode, message));
55  case DDS.RETCODE_BAD_PARAMETER.value:
56  throw new IllegalArgumentExceptionImpl(environment,
57  getErrorMessage(retCode, message));
58  case DDS.RETCODE_ERROR.value:
59  throw new DDSExceptionImpl(environment, getErrorMessage(retCode,
60  message));
61  case DDS.RETCODE_ILLEGAL_OPERATION.value:
62  throw new IllegalOperationExceptionImpl(environment,
63  getErrorMessage(retCode, message));
64  case DDS.RETCODE_IMMUTABLE_POLICY.value:
65  throw new ImmutablePolicyExceptionImpl(environment,
66  getErrorMessage(retCode, message));
67  case DDS.RETCODE_INCONSISTENT_POLICY.value:
68  throw new InconsistentPolicyExceptionImpl(environment,
69  getErrorMessage(retCode, message));
70  case DDS.RETCODE_NOT_ENABLED.value:
71  throw new NotEnabledExceptionImpl(environment, getErrorMessage(
72  retCode, message));
73  case DDS.RETCODE_UNSUPPORTED.value:
74  throw new UnsupportedOperationExceptionImpl(environment,
75  getErrorMessage(retCode, message));
76  case DDS.RETCODE_TIMEOUT.value:
77  if (withTimeOut) {
78  if (retCode == DDS.RETCODE_TIMEOUT.value) {
79  throw new TimeOutExceptionImpl(environment,
80  getErrorMessage(retCode, message));
81  }
82  }
83  case DDS.RETCODE_OK.value:
84  case DDS.RETCODE_NO_DATA.value:
85  default:
86  break;
87  }
88  }

90  private static String getDetails(DDS.ErrorInfo errorInfo, String message) {
91  String result = "";
92  DDS.StringHolder messageHolder = new DDS.StringHolder();
93 
94  errorInfo.get_message(messageHolder);
95 
96  if (messageHolder.value != null) {
97  if (message != null) {
98  result += message;
99  result += "(" + messageHolder.value + ")";
100  } else {
101  result += messageHolder.value;
102  }
103  } else if (message != null) {
104  result += message;
105  }
106  errorInfo.get_location(messageHolder);
107 
108  if (messageHolder.value != null) {
109  result += " at " + messageHolder.value;
110  }
111  errorInfo.get_source_line(messageHolder);
112 
113  if (messageHolder.value != null) {
114  result += " (" + messageHolder.value + ")";
115  }
116  errorInfo.get_stack_trace(messageHolder);
117 
118  if (messageHolder.value != null) {
119  result += "\nStack trace:\n" + messageHolder.value;
120  }
121  return result;
122  }

124  private static String getErrorMessage(int retCode, String message) {
125  String output;
126  DDS.ErrorInfo errorInfo = new ErrorInfo();
127  int result = errorInfo.update();
128 
129  switch (result) {
130  case DDS.RETCODE_NO_DATA.value:
131  output = message;
132  break;
133  case DDS.RETCODE_OK.value:
134  output = getDetails(errorInfo, message);
135  break;
136  default:
137  if (message != null) {
138  output = " Unable to get extra error information due to internal error.("
139  + message + ")";
140  } else {
141  output = " Unable to get extra error information due to internal error.";
142  }
143  break;
144  }
145  return output;
146  }

148  public static String getOsplExceptionStack(Exception ex,
149  StackTraceElement[] stack) {
150  StringBuffer result = new StringBuffer();
151  int startIndex = 0;
152 
153  result.append(ex.getClass().getSuperclass().getName() + ": "
154  + ex.getMessage() + "\n");
155 
156  while ((stack.length > startIndex)
157  && (Utilities.class.getName().equals(stack[startIndex]
158  .getClassName()))) {
159  startIndex++;
160  }
161  for (int i = startIndex; i < stack.length; i++) {
162  result.append("\tat ");
163  result.append(stack[i].getClassName());
164  result.append(".");
165  result.append(stack[i].getMethodName());
166  result.append(" (");
167  result.append(stack[i].getFileName());
168  result.append(":");
169  result.append(stack[i].getLineNumber());
170  result.append(")\n");
171  }
172  return result.toString();
173  }

181  public static void throwLastErrorException(
182  OsplServiceEnvironment environment) {
183  String message;
184  int code;
185  DDS.ErrorInfo errorInfo = new ErrorInfo();
186  int result = errorInfo.update();
187 
188  switch (result) {
189  case DDS.RETCODE_NO_DATA.value:
190  message = "";
191  code = DDS.RETCODE_ERROR.value;
192  break;
193  case DDS.RETCODE_OK.value:
194  DDS.ReturnCodeHolder errorHolder = new DDS.ReturnCodeHolder();
195  errorInfo.get_code(errorHolder);
196  code = errorHolder.value;
197  message = getDetails(errorInfo, null);
198  break;
199  default:
200  message = "Unable to get extra error information due to internal error.";
201  code = DDS.RETCODE_ERROR.value;
202  }
203 
204  switch (code) {
205  case DDS.RETCODE_PRECONDITION_NOT_MET.value:
206  throw new PreconditionNotMetExceptionImpl(environment, message);
207  case DDS.RETCODE_OUT_OF_RESOURCES.value:
208  throw new OutOfResourcesExceptionImpl(environment, message);
209  case DDS.RETCODE_ALREADY_DELETED.value:
210  throw new AlreadyClosedExceptionImpl(environment, message);
211  case DDS.RETCODE_BAD_PARAMETER.value:
212  throw new IllegalArgumentExceptionImpl(environment, message);
213  case DDS.RETCODE_ILLEGAL_OPERATION.value:
214  throw new IllegalOperationExceptionImpl(environment, message);
215  case DDS.RETCODE_IMMUTABLE_POLICY.value:
216  throw new ImmutablePolicyExceptionImpl(environment, message);
217  case DDS.RETCODE_INCONSISTENT_POLICY.value:
218  throw new InconsistentPolicyExceptionImpl(environment, message);
219  case DDS.RETCODE_NOT_ENABLED.value:
220  throw new NotEnabledExceptionImpl(environment, message);
221  case DDS.RETCODE_UNSUPPORTED.value:
222  throw new UnsupportedOperationExceptionImpl(environment, message);
223  case DDS.RETCODE_ERROR.value:
