199  public int getOldSampleState() {

201 
202  boolean read = this.sampleState.contains(SampleState.READ);
203  boolean notRead = this.sampleState.contains(SampleState.NOT_READ);
204 
205  if (read && notRead) {

273  public int getOldInstanceState() {
274  int result;
275 
276  boolean alive = this.instanceState.contains(InstanceState.ALIVE);
277  boolean disposed = this.instanceState
278  .contains(InstanceState.NOT_ALIVE_DISPOSED);
279  boolean noWriters = this.instanceState
280  .contains(InstanceState.NOT_ALIVE_NO_WRITERS);
281 
282  if (alive && disposed && noWriters) {
283  result = DDS.ANY_INSTANCE_STATE.value;
284  } else if (alive && (disposed || noWriters)) {
285  throw new IllegalArgumentExceptionImpl(this.environment,
286  "Invalid InstanceState");
287  } else if (alive) {
288  result = DDS.ALIVE_INSTANCE_STATE.value;
289  } else if (disposed && noWriters) {
290  result = DDS.NOT_ALIVE_INSTANCE_STATE.value;
291  } else if (disposed) {
292  result = DDS.NOT_ALIVE_DISPOSED_INSTANCE_STATE.value;
293  } else if (noWriters) {
294  result = DDS.NOT_ALIVE_NO_WRITERS_INSTANCE_STATE.value;
295  } else {
296  throw new IllegalArgumentExceptionImpl(this.environment,
297  "Incomplete DataState: no InstanceState set.");
298  }
299  return result;
300  }
