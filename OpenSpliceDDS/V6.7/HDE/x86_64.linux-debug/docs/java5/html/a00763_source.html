36  public DurationImpl(OsplServiceEnvironment environment, long duration,
37  TimeUnit unit) {
38  super();
39  this.environment = environment;
40 
41  if (unit == null) {
42  throw new IllegalArgumentExceptionImpl(environment,

44  }
45 
46  if (duration == Long.MAX_VALUE) {
47  this.seconds = INFINITE_SECONDS;
48  this.nanoseconds = INFINITE_NANOSECONDS;

58  public DurationImpl(OsplServiceEnvironment environment, long seconds,
59  long nanoseconds) {
60  super();
61  this.environment = environment;
62  this.seconds = seconds;
63  this.nanoseconds = nanoseconds;
64

66  && this.nanoseconds == INFINITE_NANOSECONDS) {

74  public DurationImpl normalize() {
75  long sec = this.seconds;
76  long nsec = this.nanoseconds;
77 
78  if (sec == INFINITE_SECONDS) {
79  return new DurationImpl(this.environment, INFINITE_SECONDS,
80  INFINITE_NANOSECONDS);
81  }
82  while (nsec >= 1000000000) {
83  sec += 1;
84  nsec -= 1000000000;
85 
86  if (sec == INFINITE_SECONDS) {
87  return new DurationImpl(this.environment, INFINITE_SECONDS,
88  INFINITE_NANOSECONDS);
89  }
90  }
91  return new DurationImpl(this.environment, sec, nsec);
92  }

95  public int compareTo(Duration o) {

176  @Override
177  public Duration add(Duration duration) {
178  if (duration == null) {
179  throw new IllegalArgumentException(
180  "Invalid Duration (null) provided.");
181  }
182  long sec = duration.getDuration(TimeUnit.SECONDS);
183  long nsec = duration.getRemainder(TimeUnit.SECONDS,
184  TimeUnit.NANOSECONDS);
185 
186  if (!((this.nanoseconds < 1000000000) || (this.nanoseconds == INFINITE_NANOSECONDS))) {
187  throw new IllegalArgumentExceptionImpl(this.environment,
188  "Illegal duration");
189  }
190  if (!((nsec < 1000000000) || (nsec == Long.MAX_VALUE))) {
191  throw new IllegalArgumentExceptionImpl(this.environment,
192  "Illegal duration");
193  }
194  if (this.nanoseconds == INFINITE_NANOSECONDS) {
195  if (this.seconds == INFINITE_SECONDS) {
196  return new DurationImpl(this.environment, INFINITE_SECONDS,
197  INFINITE_NANOSECONDS); /* result stays infinite. */
198  }
199  throw new IllegalArgumentExceptionImpl(this.environment,
200  "Illegal duration " + this.seconds + " seconds, "
201  + this.nanoseconds + " nanoseconds.");
202  }
203  if (nsec == Long.MAX_VALUE) {
204  if (sec == Long.MAX_VALUE) {
205  return new DurationImpl(this.environment, INFINITE_SECONDS,
206  INFINITE_NANOSECONDS); /* result stays infinite. */
207  }
208  throw new IllegalArgumentExceptionImpl(this.environment,
209  "Illegal duration " + sec + " seconds, " + nsec
210  + " nanoseconds.");
211 
212  }
213  if (sec > 0) {
214  if (INFINITE_SECONDS - sec <= this.seconds) {
215  /*
216  * is identical to 'INFINITE_SECONDS <= this.seconds + sec' In
217  * other words the sum is larger than infinite, so results must
218  * be infinite.
219  */
220  return new DurationImpl(this.environment, INFINITE_SECONDS,
221  INFINITE_NANOSECONDS);
222  }
223  }
224  long s = this.seconds + sec;
225  long ns = this.nanoseconds + nsec;
226  return new DurationImpl(this.environment, s, ns).normalize();
227 
228  }

238  @Override
239  public Duration subtract(Duration duration) {
240  if (duration == null) {
241  throw new IllegalArgumentExceptionImpl(this.environment,
242  "Invalid Duration (null) provided.");
243  }
244  if(this.isInfinite() && duration.isInfinite()){
245  throw new IllegalArgumentExceptionImpl(this.environment,
246  "Cannot subtract two infinite times.");
247  }
248  long sec = duration.getDuration(TimeUnit.SECONDS);
249  long nsec = duration.getRemainder(TimeUnit.SECONDS,
250  TimeUnit.NANOSECONDS);
251 
252 
253  if (!(this.nanoseconds < 1000000000)
254  || (this.nanoseconds == INFINITE_NANOSECONDS)) {
255  throw new IllegalArgumentExceptionImpl(this.environment,"Illegal duration "
256  + this.seconds + " seconds, " + this.nanoseconds
257  + " nanoseconds.");
258  }
259  if (!((nsec < 1000000000) || (nsec == Long.MAX_VALUE))) {
260  throw new IllegalArgumentExceptionImpl(this.environment,"Illegal duration " + sec
261  + " seconds, " + nsec + " nanoseconds.");
262  }
263 
264  if (this.nanoseconds == INFINITE_NANOSECONDS) {
265  if (this.seconds == INFINITE_SECONDS) {
266  return new DurationImpl(this.environment, INFINITE_SECONDS,
267  INFINITE_NANOSECONDS); /* result stays infinite. */
268  }
269  throw new IllegalArgumentExceptionImpl(this.environment,
270  "Illegal duration " + this.seconds + " seconds, "
271  + this.nanoseconds + " nanoseconds.");
272  }
273  if (nsec == Long.MAX_VALUE) {
274  if (sec == Long.MAX_VALUE) {
275  return new DurationImpl(this.environment, INFINITE_SECONDS,
276  INFINITE_NANOSECONDS); /* result stays infinite. */
277  }
278  throw new IllegalArgumentExceptionImpl(this.environment,
279  "Illegal duration " + sec + " seconds, " + nsec
280  + " nanoseconds.");
281  }
282  if ((sec <= 0) && (INFINITE_SECONDS + sec <= this.seconds)) {
283  /* is identical to '-(INFINITE_SECONDS >= this.seconds - sec)' In
284  * other words the sum is larger than infinite, so results must be
285  * infinite. */
286  return new DurationImpl(this.environment, INFINITE_SECONDS,
287  INFINITE_NANOSECONDS);
288  }
289  long s = this.seconds - sec;
290 
291  if (s == INFINITE_SECONDS) {
292  return new DurationImpl(this.environment, INFINITE_SECONDS,
293  INFINITE_NANOSECONDS);
294  }
295  long ns = this.nanoseconds - nsec;
296 
297  if (ns < 0) {
298  s--;
299  ns += 1000000000;
300  }
301  return new DurationImpl(this.environment, s, ns).normalize();
302 
303  }
